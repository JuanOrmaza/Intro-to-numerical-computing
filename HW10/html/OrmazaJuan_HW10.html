    exact_integral=pi;
    n=[4,8,16,32];
    p=zeros(1,length(n));
    error = zeros(1,length(n));
    h = zeros(1,length(n));
    integral_results = zeros(1,length(n));

    for i=1:length(n)
        integral_results(i) =  comp_gauss_int(f,0,pi,n(i));
        error(i) = abs(exact_integral-comp_gauss_int(f,0,pi,n(i)));
        h(i)=pi/n(i);
        if(i>=2)
            p(i)=log(error(i)/error(i-1))/log(h(i)/h(i-1));
        end
    end
